15static void exec_external_cmd(t_cmd *curr, t_shell *shell)
16{
17 char *cmd_path;
18 char **envp;
19
20 if (handle_redirections(curr) < 0)
21 exit(1);
22 if (!curr->args || !curr->args[0])
23 exit(0);
24 cmd_path = find_command_path(curr->args[0], shell->env);
25 if (!cmd_path)
26 {
27 ft_putstr_fd(curr->args[0], 2);
28 ft_putstr_fd(ERR_CMD_NOT_FOUND, 2);
29 exit(127);
30 }
31 envp = env_to_array(shell->env);

37}

40{

58static pid_t launch_child(t_cmd *curr, int *prev_fd, t_shell *shell)
59{
60 int pipefd[2];
61 pid_t pid;
62
63 if (curr->next && pipe(pipefd) == -1)
64 return (-1);
65 pid = fork();
66 if (pid == 0)
67 child_process(curr, *prev_fd, pipefd, shell);
68 if (pid < 0)
69 {
70 perror("fork");
71 return (-1);
72 }
73 if (*prev_fd != -1)
74 close(*prev_fd);
75 if (curr->next)
76 {
77 close(pipefd[1]);
78 *prev_fd = pipefd[0];
79 }
80 return (pid);
81}

83static void wait_for_all(pid_t *pids, int n, t_shell *shell)
84{
85 int status;
86 pid_t pid;
87 int i;
88 pid_t last_pid;
89
90 last_pid = pids[n - 1];
91 i = 0;
92 while (i < n)
93 {
94 pid = waitpid(-1, &status, 0);
95 if (pid == last_pid)
96 {
97 if (WIFEXITED(status))
98 shell->exit_status = WEXITSTATUS(status);
99 else if (WIFSIGNALED(status))
100 {
101 int sig = WTERMSIG(status);
102 if (sig == SIGQUIT)
103 ft_putstr_fd("Quit: 3\n", STDERR_FILENO);
104 shell->exit_status = 128 + sig;
105 }
106 else
107 shell->exit_status = 1;
108 }
109 i++;
110 }
111}

113void execute_piped_commands(t_cmd *cmd_list, t_shell *shell)
114{
115 t_cmd *curr;
