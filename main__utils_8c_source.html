62void handle_commands(char *line, t_shell *shell)
63{
64 t_tkn *tokens;
65 t_cmd *cmds;
66 int stdin_backup;
67 int stdout_backup;
68
69 LOG_DEBUG("Handling input: [%s]", line);
70 stdin_backup = dup(STDIN_FILENO);
71 stdout_backup = dup(STDOUT_FILENO);
72 if (!shell || !line)
73 return ;
74 tokens = prepare_tokens(line, shell);
75 if (!tokens)
76 {
77 LOG_WARN("Tokenization failed → setting shell->exit_status = 2\n");
78 shell->exit_status = 2;
79 return ;
80 }
81 cmds = check_and_parse(tokens, shell);
82 if (!cmds)
83 {
84 LOG_WARN("Parsing failed → setting shell->exit_status = 2\n");
85 shell->exit_status = 2;
86 return ;
87 }
88 shell->cmds = cmds;
89 handle_heredoc(cmds, shell);
90 LOG_DEBUG("Calling execute_commands()\n");
91 execute_commands(cmds, shell, line);
92 restore_stdio(stdin_backup, stdout_backup);
93 ft_free_tokens(tokens);
94 free_cmd_list(cmds);
95 shell->tkns = NULL;
96 shell->cmds = NULL;
97}
