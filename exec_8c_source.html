28static int wait_for_process(pid_t pid, char *cmd_path)

30 int status;

51int exec_cmd(char *cmd, char **args, t_env *env)
52{
53 char *cmd_path;
54 pid_t pid;
55 int status;
56
57 cmd_path = find_command_path(cmd, env);
58 if (!cmd_path)
59 return (ft_putstr_fd(cmd, 2), ft_putstr_fd(ERR_CMD_NOT_FOUND, 2), 127);
60 signal(SIGINT, SIG_IGN);
61 signal(SIGQUIT, SIG_IGN);
62 pid = fork();
63 if (pid < 0)
64 return (perror("fork"), free(cmd_path), 1);
65 if (pid == 0)
66 {
67 reset_signals();
68 execute_child_process(cmd_path, args, env);
69 }
70 else
71 {
72 status = wait_for_process(pid, cmd_path);
73 }
74 set_signals();
75 return (status);
76}
