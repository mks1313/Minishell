68# Dependency for libft now includes this Makefile

73# To create the final executable

78# To compile source files to object files
79$(OBJ_DIR)/%.o: $(SRC_PATH)/%.c $(INC_H) Makefile
80 @if [ ! -d $(dir $@) ]; then \
81 mkdir -p $(dir $@); \

83 $(CC) $(CFLAGS) $(SANITIZE) -c $< -o $@
