15static void update_exit_status(int status, t_shell *shell)
16{
17 int sig;
18
19 if (WIFEXITED(status))
20 shell->exit_status = WEXITSTATUS(status);
21 else if (WIFSIGNALED(status))
22 {
23 sig = WTERMSIG(status);
24 if (sig == SIGQUIT)
25 ft_putstr_fd("Quit: 3\n", STDERR_FILENO);
26 shell->exit_status = 128 + sig;
27 g_exit_status = 128 + sig;
28 }
29 else
30 {
31 shell->exit_status = 1;
32 g_exit_status = 1;
33 }
34}
