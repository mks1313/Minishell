15int is_valid_identifier(const char *key)
16{
17 int i;
18
19 if (!key || (!ft_isalpha(key[0]) && key[0] != '_'))
20 {
21 ft_printf("export: `%s`: not a valid identifier\n", key);
22 return (0);
