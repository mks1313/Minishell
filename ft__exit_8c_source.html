49void ft_exit(t_cmd *cmd, t_shell *shell)
50{
51 char **args;
52 int arg_count;
53
54 args = cmd->args;
55 arg_count = 0;
56 while (args && args[arg_count])
57 arg_count++;
58 if (arg_count == 1)
59 exit_shell(shell, EXIT_SUCCESS);
60 if (args[1] && !is_numeric_argument(args[1]))
61 {
62 print_numeric_error(args[1]);
63 exit_shell(shell, 255);
64 }
65 if (arg_count > 2)
66 {
67 ft_putstr_fd(ERR_EXIT_TOO_MANY_ARGS, STDERR_FILENO);
68 return ;
69 }
70 exit_shell(shell, (unsigned char)ft_atol(args[1]));
71}
