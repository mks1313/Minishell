15int validate_syntax(t_cmd *cmd_list, t_shell *shell)

17 t_cmd *cmd;

53static bool validate_redirect(t_tkn *token, t_shell *shell)
54{
55 t_tkn *next;
56
57 next = token->next;
58 if (!next)

65bool validate_token_syntax(t_tkn *tokens, t_shell *shell)
66{
67 t_tkn *prev;
68
69 prev = NULL;
70 while (tokens)
71 {
72 if (tokens->type == TOK_PIPE)
73 {
74 if (!validate_pipe(prev, tokens, shell))
75 return (false);
76 }
77 else if (tokens->type >= TOK_REDIR_IN && tokens->type <= TOK_HEREDOC)
78 {
79 if (!validate_redirect(tokens, shell))
80 return (false);
81 }
82 else if (tokens->type == TOK_OPERATOR)
83 {
84 return (syntax_error(shell, NULL, TOK_OPERATOR));
85 }
86 prev = tokens;
87 tokens = tokens->next;
88 }
89 return (true);
90}
